215//*****************************************************************************
216// Functions to carry out the initialization of RW and BSS data sections. These
217// are written as separate functions rather than being inlined within the
218// ResetISR() function in order to cope with MCUs with multiple banks of
219// memory.
220//*****************************************************************************
221__attribute__ ((section(".after_vectors")))
222void data_init(unsigned int romstart, unsigned int start, unsigned int len) {
223 unsigned int *pulDest = (unsigned int*) start;
224 unsigned int *pulSrc = (unsigned int*) romstart;
225 unsigned int loop;
226 for (loop = 0; loop < len; loop = loop + 4)
227 *pulDest++ = *pulSrc++;
228}
229
230__attribute__ ((section(".after_vectors")))
231void bss_init(unsigned int start, unsigned int len) {
232 unsigned int *pulDest = (unsigned int*) start;
233 unsigned int loop;
234 for (loop = 0; loop < len; loop = loop + 4)
235 *pulDest++ = 0;
236}

238//*****************************************************************************
239// The following symbols are constructs generated by the linker, indicating
240// the location of various points in the "Global Section Table". This table is
241// created by the linker via the Code Red managed linker script mechanism. It
242// contains the load address, execution address and length of each RW data
243// section and the execution and length of each BSS (zero initialized) section.
244//*****************************************************************************
245extern unsigned int __data_section_table;
246extern unsigned int __data_section_table_end;
247extern unsigned int __bss_section_table;
248extern unsigned int __bss_section_table_end;

250//*****************************************************************************
251// Reset entry point for your code.
252// Sets up a simple runtime environment and initializes the C/C++
253// library.
254//*****************************************************************************
255__attribute__ ((section(".after_vectors")))
256void
257ResetISR(void) {
258
259 //
260 // Copy the data sections from flash to SRAM.
261 //
262 unsigned int LoadAddr, ExeAddr, SectionLen;
263 unsigned int *SectionTableAddr;
264
265 // Load base address of Global Section Table
266 SectionTableAddr = &__data_section_table;
267
268 // Copy the data sections from flash to SRAM.
269 while (SectionTableAddr < &__data_section_table_end) {
270 LoadAddr = *SectionTableAddr++;
271 ExeAddr = *SectionTableAddr++;
272 SectionLen = *SectionTableAddr++;
273 data_init(LoadAddr, ExeAddr, SectionLen);
274 }
275 // At this point, SectionTableAddr = &__bss_section_table;
276 // Zero fill the bss segment
277 while (SectionTableAddr < &__bss_section_table_end) {
278 ExeAddr = *SectionTableAddr++;
279 SectionLen = *SectionTableAddr++;
280 bss_init(ExeAddr, SectionLen);
281 }
